## 20. 有效的括号

```C++
class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        for(char c : s){
            if(c == '(') st.push(')');
            else if(c == '{') st.push('}');
            else if(c == '[') st.push(']');
            else{
                if(st.empty()) return false;
                if(c == st.top()) st.pop();
                else return false;
            }
        }
        if(!st.empty()) return false;
        return true;
    }
};
```
