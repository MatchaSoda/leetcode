## 35. 搜索插入位置

```C++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        int left = 0;
        int right = n - 1;
        while (left <= right) {
            int middle = (left + right) / 2;
            if (nums[middle] > target) {
                right = middle - 1;//right右边的都比target大
            } else if (nums[middle] < target) {
                left = middle + 1;//left左边的都比target小
            } else {
                return middle;
            }
        }
        //若没有找到target，结果一定是right在left左边一格
        return left;//等同于return right + 1;    
    }
};
```
