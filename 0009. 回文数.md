## 9. 回文数

```C++
class Solution {
public:
    bool isPalindrome(int x) {
        if(x < 0) return false;
        if(x < 10) return true;
        long long pre = x;
        long long cur = 0;
        while(x){
            cur = cur * 10 + x % 10;
            x /= 10;
        }
        return pre == cur;
    }
};
```
